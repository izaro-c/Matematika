      onMouseEnter={() => setVariable('highlight', target)}
      onMouseLeave={() => setVariable('highlight', null)}
    >
      {children}
    </span>
  );
};

<div className="pt-24 pb-8 mb-8 max-w-prose mx-auto text-center">
# Teorema de Pitágoras

<DemonstrationSection diagram={<RearrangementStep1 />}>

<MedievalStep number={1} title="El Cuadrado y los Triángulos" />

Construyamos un gran cuadrado exterior. Para ello, alineamos cuatro copias idénticas de nuestro triángulo rectángulo de modo que los catetos se empalmen formando una línea recta: el <InteractiveElement target="ladoA" color="terracota">lado a</InteractiveElement> de uno junto al <InteractiveElement target="ladoB" color="terracota">lado b</InteractiveElement> del siguiente. 

Al cerrar el perímetro, obtenemos un inmenso cuadrado cuyos lados miden exactamente **(a+b)**. Por definición geométrica, el área total de este lienzo exterior debe ser invariablemente **(a+b)²**.

En su interior, nuestros <InteractiveElement target="triangles" color="terracota">cuatro triángulos</InteractiveElement> se asientan en las esquinas. El espacio vacío que dejan en el centro está perfectamente delineado por las cuatro hipotenusas. Como cada triángulo aporta su <InteractiveElement target="ladoC" color="salvia">lado c</InteractiveElement>, el hueco central es un cuadrado perfecto de área <InteractiveElement target="areaC" color="salvia"><span className="font-serif">c²</span></InteractiveElement>.

Si sumamos las áreas de las piezas individuales que componen el dibujo, obtenemos la primera ecuación para nuestra área total:

<div className="mt-8 p-6 bg-lienzo border-2 border-carbon/10 shadow-sm rounded-lg font-serif">
  <p className="font-bold text-carbon mb-2">Cálculo del Área Total:</p>
  <p>Área Total = <span className="italic text-salvia">Área del Vacío Central</span> + <span className="italic text-terracota">Área de 4 triángulos</span></p>
  <p className="text-2xl text-center mt-4">
    <span className="font-bold text-salvia">c²</span> + <span className="text-terracota">4 × (ab / 2)</span>
  </p>
</div>

</DemonstrationSection>

<DemonstrationSection diagram={<RearrangementStep2 />}>

<MedievalStep number={2} title="La Reorganización" />

Aplicamos una traslación geométrica a los <InteractiveElement target="triangles" color="terracota">cuatro triángulos</InteractiveElement> dentro del mismo cuadrado exterior, agrupándolos de dos en dos para formar dos rectángulos de lados *a* y *b*.

Al reorganizar el interior sin modificar el área total, el espacio vacío original queda particionado en dos nuevos cuadrados. 

Dado que conocemos los catetos de los triángulos, determinamos que uno de los huecos resultantes tiene lado *a* (por lo tanto, área <InteractiveElement target="areaA" color="salvia"><span className="font-serif">a²</span></InteractiveElement>) y el otro tiene lado *b* (área <InteractiveElement target="areaB" color="salvia"><span className="font-serif">b²</span></InteractiveElement>).

<div className="mt-8 p-6 bg-lienzo border-2 border-carbon/10 shadow-sm rounded-lg font-serif">
  <p className="font-bold text-carbon mb-2">Cálculo del Área Total:</p>
  <p>Área Total = <span className="italic text-salvia">Área de los Nuevos Vacíos</span> + <span className="italic text-terracota">Área de 4 triángulos</span></p>
  <p className="text-2xl text-center mt-4">
    <span className="font-bold text-salvia">a² + b²</span> + <span className="text-terracota">4 × (ab / 2)</span>
  </p>
</div>

</DemonstrationSection>

<DemonstrationSection diagram={<RearrangementConclusion />}>

<MedievalStep number={3} title="La Deducción Final" />

Dado que el cuadrado exterior de lado (a+b) es el mismo en ambas figuras (su área total nunca cambia), y los cuatro triángulos que hemos restado de su interior son los mismos, **el espacio vacío sobrante debe ser obligatoriamente igual** en ambas configuraciones. 

Igualamos las dos ecuaciones de área que acabamos de calcular:

<div className="mt-8 mb-4 p-8 bg-lienzo border-2 border-carbon/10 shadow-sm rounded-lg font-serif text-center text-xl">
  <div className="mb-4">
    <span className="italic text-pizarra text-sm block mb-1">Área Paso 1 = Área Paso 2</span>
    <span className="font-bold text-salvia">c²</span> + <span className="text-terracota">4(ab/2)</span> = <span className="font-bold text-salvia">a² + b²</span> + <span className="text-terracota">4(ab/2)</span>
  </div>
  
  <div className="mb-4 text-carbon/40 font-bold">
    &darr;
  </div>
  
  <div className="mb-4">
    <span className="italic text-pizarra text-sm block mb-1">Restamos los 4 triángulos a ambos lados</span>
    <span className="font-bold text-salvia">c²</span> + <span className="text-terracota line-through opacity-50">4(ab/2)</span> = <span className="font-bold text-salvia">a² + b²</span> + <span className="text-terracota line-through opacity-50">4(ab/2)</span>
  </div>

  <div className="mb-4 text-carbon/40 font-bold">
    &darr;
  </div>

  <div className="mt-6 text-4xl">
    <span className="text-salvia font-bold">c²</span> = <span className="text-terracota font-bold">a² + b²</span>
  </div>
</div>

La verdad matemática se destila pura e irrefutable. El área del cuadrado construido sobre la hipotenusa es exactamente igual a la suma de las áreas de los cuadrados construidos sobre los catetos.
