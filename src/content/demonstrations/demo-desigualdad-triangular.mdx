export const Component = () => {

  return (
    <DemonstrationSection diagrams={diagrams}>
      <MedievalStep number={1} target={["lado-ab", "lado-bc", "lado-ac"]} title="El triángulo base">
        Consideremos un triángulo con vértices $A, B, C$. Sus longitudes laterales son <InteractiveElement target="lado-ab">$c$</InteractiveElement>, <InteractiveElement target="lado-bc">$a$</InteractiveElement> y <InteractiveElement target="lado-ac">$b$</InteractiveElement>. Queremos demostrar que $b + a \gt c$.
      </MedievalStep>
      
      <MedievalStep number={2} target={["lado-cd", "punto-d"]} title="Extensión del lado">
        Prolonguemos el lado $AC$ más allá de $C$ hasta un nuevo punto <InteractiveElement target="punto-d">$D$</InteractiveElement>, de manera que el <InteractiveElement target="lado-cd">segmento $CD$</InteractiveElement> tenga exactamente la misma longitud que el lado $BC$ (es decir, $CD = a$).
      </MedievalStep>

      <MedievalStep number={3} target={["triangulo-isosceles", "angulo-cbd", "angulo-cdb"]} title="Triángulo isósceles auxiliar">
        Trazamos el <InteractiveElement target="lado-bd">segmento $BD$</InteractiveElement>. Al hacerlo, hemos formado el triángulo $\triangle BCD$. Como por construcción $BC = CD = a$, este es un <InteractiveElement target="triangulo-isosceles">triángulo isósceles</InteractiveElement>. Por lo tanto, los ángulos opuestos a estos lados iguales también son iguales: <InteractiveElement target="angulo-cbd">$\angle CBD$</InteractiveElement> $=$ <InteractiveElement target="angulo-cdb">$\angle CDB$</InteractiveElement>.
      </MedievalStep>

      <MedievalStep number={4} target={["angulo-abd", "angulo-cdb"]} title="Relación de ángulos mayores">
        Observemos el <InteractiveElement target="angulo-abd">ángulo completo $\angle ABD$</InteractiveElement>. Es obvio que $\angle ABD = \angle ABC + \angle CBD$. Por lo tanto, $\angle ABD \gt \angle CBD$. Y como sabíamos del paso anterior que $\angle CBD = \angle CDB$, deducimos que el <InteractiveElement target="angulo-abd">$\angle ABD$</InteractiveElement> es estrictamente mayor que el <InteractiveElement target="angulo-cdb">$\angle CDB$</InteractiveElement> (que es el mismo que $\angle ADB$).
      </MedievalStep>

      <MedievalStep number={5} target={["lado-ab", "lado-ac", "lado-cd"]} title="Lado mayor se opone a ángulo mayor">
        En todo triángulo (en este caso el $\triangle ABD$), al ángulo mayor se le opone el lado de mayor longitud. Como $\angle ABD \gt \angle ADB$, el lado opuesto a $\angle ABD$ (que es todo el segmento $AD$) debe ser mayor que el lado opuesto a $\angle ADB$ (que es el <InteractiveElement target="lado-ab">segmento $AB = c$</InteractiveElement>). Es decir, $AD \gt c$.
        Pero $AD$ se compone de <InteractiveElement target="lado-ac">$AC (b)$</InteractiveElement> y <InteractiveElement target="lado-cd">$CD (a)$</InteractiveElement>. Por tanto, $b + a \gt c$. $\blacksquare$
      </MedievalStep>
    </DemonstrationSection>
  );
