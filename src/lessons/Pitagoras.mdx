import { useMathStore } from '../store/MathStoreContext';
import { Link } from 'wouter';

export const InteractiveFormulaElement = ({ target, color, symbol }) => {
  const setVariable = useMathStore(state => state.setVariable);
  
  return (
    <span 
      style={{ color: `var(--color-${color})` }} 
      className="font-bold cursor-pointer hover:bg-carbon/5 px-2 py-1 rounded-md transition-colors inline-block"
      onMouseEnter={() => setVariable('highlight', target)}
      onMouseLeave={() => setVariable('highlight', null)}
    >
      {symbol}²
    </span>
  );
};

export const MathVariable = ({ name, color }) => {
  const value = useMathStore(state => state.variables[name] || 0);
  return <span style={{ color: `var(--color-${color})` }} className="font-bold">{value.toFixed(2)}</span>;
};

# El Teorema de Pitágoras

Atribuido a <ConceptLink to="bio/pitagoras">Pitágoras de Samos</ConceptLink> y su mística hermandad, este teorema establece que en todo triángulo rectángulo, el cuadrado de la hipotenusa es igual a la suma de los cuadrados de los catetos.

Pasa el ratón por encima de las letras de la fórmula para visualizar su correspondencia geométrica en el lienzo izquierdo:

<div className="text-4xl text-center my-12 p-8 bg-lienzo rounded-xl shadow-sm border border-carbon/10 font-serif flex justify-center items-center gap-4">
  <InteractiveFormulaElement target="areaA" color="terracota" symbol="a" /> + 
  <InteractiveFormulaElement target="areaB" color="terracota" symbol="b" /> = 
  <InteractiveFormulaElement target="areaC" color="salvia" symbol="c" />
</div>

Arrastra los vértices en el simulador para deformarlo. Las áreas de los cuadrados (catetos al cuadrado y la hipotenusa al cuadrado) se mantendrán vivas calculando este equilibrio numérico perfecto:

<div className="text-xl text-center my-8 p-4 bg-lienzo/50 rounded-lg font-serif">
  <MathVariable name="catetoA" color="terracota" />² + <MathVariable name="catetoB" color="terracota" />² = <MathVariable name="hipotenusa" color="salvia" />²
</div>
